Here it is:

def nearest_value(values: set[int], one: int) -> int:

    min_dist, closest_num  = None, None
    for value in values:
        dist = abs(one - value)
        if min_dist == None or min_dist > dist:
            min_dist, closest_num = dist, value
        elif min_dist == dist:
            closest_num = min(value, closest_num)
        
    return closest_num

Much shorter solution, where you find minimum value in set by two parameters: first is distance (absolute difference) from the input number, second: smaller value between two on the same distance 

def nearest_value(values: set[int], one: int) -> int:

    return min(values, key=lambda n: (abs(one - n), n))

In this solution you start from one and increasing and decreasing it check if this new value is in set. 

def nearest_value(values: set[int], one: int) -> int:

    n = 0
    while True:
        if (one - n) in values:
            return one - n
        if (one + n) in values:
            return one + n
        n += 1
