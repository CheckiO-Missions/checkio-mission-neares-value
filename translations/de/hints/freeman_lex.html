Hier ist sie:

def nearest_value(values: set[int], one: int) -> int:

    min_dist, closest_num  = None, None
    for value in values:
        dist = abs(one - value)
        if min_dist == None or min_dist> dist:
            min_dist, closest_num = dist, value
        elif min_dist == dist:
            closest_num = min(value, closest_num)
        
    return closest_num

Viel kürzere Lösung, bei der Sie den Mindestwert in der Menge durch zwei Parameter finden: erstens den Abstand (absolute Differenz) von der eingegebenen Zahl, zweitens den kleineren Wert zwischen zwei auf dem gleichen Abstand 

def nearest_value(values: set[int], one: int) -> int:

    return min(values, key=lambda n: (abs(one - n), n))

Bei dieser Lösung gehen Sie von one aus und prüfen durch Erhöhen und Verringern, ob der neue Wert in der Menge enthalten ist. 

def nearest_value(values: set[int], one: int) -> int:

    n = 0
    while True:
        if (one - n) in values:
            return one - n
        if (one + n) in values:
            return one + n
        n += 1
