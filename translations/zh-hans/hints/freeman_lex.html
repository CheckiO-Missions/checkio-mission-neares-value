def nearest_value(values: set[int], one: int) -> int:

    min_dist, closest_num  = None, None
    for value in values:
        dist = abs(one - value)
        if min_dist == None or min_dist> dist:
            min_dist, closest_num = dist, value
        elif min_dist == dist:
            closest_num = min(value, closest_num)
        
    return closest_num

更简短的解决方案，通过两个参数找到集合中的最小值：第一个参数是与输入数字的距离（绝对差值），第二个参数是相同距离的两个数字之间的较小值。 

def nearest_value(values: set[int], one: int) -> int:

    return min(values, key=lambda n: (abs(one - n), n))

在此方案中，您可以从one 开始，增加或减少它，检查新值是否在集合中。 

def nearest_value(values: set[int], one: int) -> int:

    n = 0
    while True:
        if (one - n) in values:
            return one - n
        if (one + n) in values:
            return one + n
        n += 1
