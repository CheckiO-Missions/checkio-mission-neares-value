def nearest_value(values: set[int], one: int) -> int:

    min_dist, closest_num  = None, None
    for value in values:
        dist = abs(one - value)
        if min_dist == None or min_dist > dist:
            min_dist, closest_num = dist, value
        elif min_dist == dist:
            closest_num = min(value, closest_num)
        
    return closest_num

Набагато коротший розв'язок, в якому ти знаходиш мінімальне значення в множині за двома параметрами: перший - це відстань (абсолютна різниця) від вхідного числа, другий: менше значення з двох на однаковій відстані. 

def nearest_value(values: set[int], one: int) -> int:

    return min(values, key=lambda n: (abs(one - n), n))
У цьому розв'язок ти починаєш з вхідного числа "one", збільшуєш і зменшуєш його і перевіряєш, чи є це нове значення у множині. 

def nearest_value(values: set[int], one: int) -> int:

    n = 0
    while True:
        if (one - n) in values:
            return one - n
        if (one + n) in values:
            return one + n
        n += 1
